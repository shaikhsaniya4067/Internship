10) creating external table:
1) CREATE EXTERNAL TABLE IF NOT EXISTS
external_airports (
    airport_code STRING,
    airport_name STRING,
    city STRING,
    state STRING
)
ROW FORMAT DELIMITED
FIELDS TERMINATED BY ','
STORED AS TEXTFILE;

2) Join:

SELECT f.flight_num, f.origin, f.dest,
       e.airport_name, e.city, e.state
FROM flights f
JOIN external_airports e
ON (f.dest = e.airport_code);
